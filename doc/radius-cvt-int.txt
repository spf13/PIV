  int radius_cvt_int(string data)~

<?php >
  while ($resa = radius_get_attr($res)) {
  
      if (!is_array($resa)) {
          printf ("Error getting attribute: %s\n",  radius_strerror($res));
          exit;
      }
  
      $attr = $resa['attr'];
      $data = $resa['data'];
      
      switch ($attr) {
  
      case RADIUS_FRAMED_MTU:
          $mtu = radius_cvt_int($data);
          echo "MTU: $mtu<br>\n";
          break;
      }
  }
?>
